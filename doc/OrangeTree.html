def oneYearPasses
  @age = @age + 1
  @height = @height + 50

  if alive?
    puts "Another year goes by..."
    puts "...but your orange tree has run its course and has passed away."
    exit
  end

  puts "One year goes by. Your orange tree is growing nicely."

  if fruit?
    @fruit = 10
    puts "Your orange tree has produced " + @fruit.to_s + " oranges."
  end


end
